import pytest


class TestKDSSingleMenuServeFlow:

    @pytest.mark.smoke
    def test_single_menu_served_flow(
            self,
            create_order,
            serve_single_menu
    ):

        # Step 1 Create Order
        create_res = create_order()

        order_id = create_res["order_id"]

        print("Order ID:", order_id)

        assert order_id is not None


        # Step 2 Serve Menu
        update_res = serve_single_menu(order_id)


        # Step 3 Validate response
        assert update_res["order_id"] == order_id


        # Step 4 Validate message
        assert update_res["detail"] == "Order status updated to served"
